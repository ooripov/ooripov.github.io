  // ===== Helpers =====
  function getSavedLang() {
    const saved = localStorage.getItem('lang');
    if (saved && (saved === 'ru' || saved === 'en')) return saved;
    const nav = (navigator.language || 'ru').toLowerCase();
    return nav.startsWith('en') ? 'en' : 'ru';
  }

  function buildMapUrl(q) {
    const query = encodeURIComponent(q || 'Khujand');
    return 'https://www.google.com/maps/search/?api=1&query=' + query;
  }

  function i18nKey(lang, key, fallback) {
    return (I18N[lang] && I18N[lang][key]) || fallback || '';
  }

  function normalizePhone(p) {
    return String(p || '').replace(/\s|\(|\)|-/g, '');
  }

  async function copyText(text) {
    try {
      await navigator.clipboard.writeText(text);
      return true;
    } catch {
      const ta = document.createElement('textarea');
      ta.value = text;
      ta.style.position = 'fixed';
      ta.style.left = '-9999px';
      document.body.appendChild(ta);
      ta.select();
      const ok = document.execCommand('copy');
      document.body.removeChild(ta);
      return ok;
    }
  }

  function pickLangText(item, lang, base) {
    if (base === 'name') return (lang === 'en' ? item.name_en : item.name_ru) || item.name_ru || item.name_en || '';

  function createPlaceCard(item, lang) {
    const name = pickLangText(item, lang, 'name');
    const desc = pickLangText(item, lang, 'desc');

    const card = document.createElement('div');
    card.className = 'place-card';

    const head = document.createElement('div');
    head.className = 'place-head';

    const left = document.createElement('div');
    const h = document.createElement('div');
    h.className = 'place-title';
    h.textContent = name || '‚Äî';

    const p = document.createElement('div');
    p.className = 'place-sub';
    p.textContent = desc || '';

    left.appendChild(h);
    if (desc) left.appendChild(p);

    head.appendChild(left);
    card.appendChild(head);

    const meta = document.createElement('div');
    meta.className = 'place-meta';

    const addRow = (k, v) => {
      if (!v) return;
      const row = document.createElement('div');
      row.className = 'meta-row';
      const kk = document.createElement('div');
      kk.className = 'meta-k';
      kk.textContent = k;
      const vv = document.createElement('div');
      vv.textContent = v;
      row.appendChild(kk);
      row.appendChild(vv);
      meta.appendChild(row);
    };

    addRow(i18nKey(lang, 'field_address', 'Address'), item.address);
    addRow(i18nKey(lang, 'field_hours', 'Hours'), item.hours);
    addRow(i18nKey(lang, 'field_phone', 'Phone'), item.phone);

    if (meta.childElementCount) card.appendChild(meta);

    const actions = document.createElement('div');
    actions.className = 'place-actions';

    const phone = normalizePhone(item.phone);
    const whatsapp = normalizePhone(item.whatsapp);

    if (item.mapQuery || item.address) {
      const a = document.createElement('a');
      a.className = 'pill';
      a.target = '_blank';
      a.rel = 'noopener';
      a.href = buildMapUrl(item.mapQuery || item.address);
      a.textContent = i18nKey(lang, 'btn_map', 'Map');
      actions.appendChild(a);
    }

    if (whatsapp) {
      const wa = whatsapp.startsWith('+') ? whatsapp.slice(1) : whatsapp;
      const a = document.createElement('a');
      a.className = 'pill';
      a.target = '_blank';
      a.rel = 'noopener';
      a.href = 'https://wa.me/' + wa;
      a.textContent = i18nKey(lang, 'btn_whatsapp', 'WhatsApp');
      actions.appendChild(a);
    }

    if (phone) {
      const a = document.createElement('a');
      a.className = 'pill';
      a.href = 'tel:' + phone;
      a.textContent = i18nKey(lang, 'btn_call', 'Call');

      b.className = 'pill';

  // ===== Language apply (instant) =====
  function applyLangInstant(lang){
    if (!I18N[lang]) return;

    localStorage.setItem('lang', lang);
    document.documentElement.lang = lang;

    const titleEl = document.querySelector('[data-i18n-title="doc_title"]');
    if (titleEl) titleEl.textContent = I18N[lang].doc_title;

    document.querySelectorAll('[data-i18n]').forEach(el => {
      const key = el.getAttribute('data-i18n');
      const value = I18N[lang][key];
      if (typeof value === 'string') el.textContent = value;
    });

    // toggle buttons state (both places)
    document.querySelectorAll('.lang-btn').forEach(btn => {
      const isActive = btn.getAttribute('data-lang') === lang;
      btn.setAttribute('aria-pressed', isActive ? 'true' : 'false');
    });

    // rerender dynamic current page
    const current = (location.hash || '#home').slice(1) || 'home';
    if (current && current !== 'splash') {
      renderPageContent(current);
      if (current === 'journal-modern') renderModernSpots();
      if (current === 'journal-old') renderOldSpots();
      if (current === 'gift-store') renderGifts();
    }

    // if modal open, refresh text by language
    if (modalState && modalState.open) {
      if (modalState.spotId) openPlaceModal(modalState.spotId);
      if (modalState.giftId) openGiftModal(modalState.giftId);
    }
  }

  // ===== Navigation + language (smooth) =====
  let langSwitchTimer = null;

  function setLang(lang) {
    if (!I18N[lang]) return;

    if (langSwitchTimer) clearTimeout(langSwitchTimer);

    // fade out
    document.body.classList.add('lang-fade');

    langSwitchTimer = setTimeout(() => {
      applyLangInstant(lang);
      showPage(); // update hero title/bg, render, etc.

      // fade in
      requestAnimationFrame(() => {
        document.body.classList.remove('lang-fade');
      });
    }, 170);
  }

  function showPage(id) {
    const lang = getSavedLang();
    const pageId = id || (location.hash || '#home').slice(1) || 'home';

    if (pageId === 'splash') {
      document.body.classList.add('splash-active');
      document.body.classList.remove('splash-closing');
      const splash = document.getElementById('splash');
      if (splash) splash.classList.remove('closing');
      const hint = document.querySelector('.splash-hint');
      if (hint) hint.textContent = I18N[lang].tap_to_start || 'Tap to start';
      return;
    }

    document.body.classList.remove('splash-active');

    document.querySelectorAll('.page').forEach(p => p.classList.add('hidden'));
    (document.getElementById(pageId) || document.getElementById('home')).classList.remove('hidden');

    const h1 = document.querySelector('.hero h1');
    const pageTitle = (PAGE_TITLES[lang] && PAGE_TITLES[lang][pageId]) || (PAGE_TITLES[lang] && PAGE_TITLES[lang].home) || 'Welcome!';
    if (h1) h1.textContent = pageTitle;

    const hero = document.querySelector('.hero');
    if (hero) hero.style.setProperty('--hero-bg', heroBgMap[pageId] || heroBgMap.home);

    renderPageContent(pageId);
    if (pageId === 'journal-modern') renderModernSpots();
    if (pageId === 'journal-old') renderOldSpots();
    if (pageId === 'gift-store') renderGifts();

    const active = document.getElementById(pageId) || document.getElementById('home');
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    requestAnimationFrame(() => { if (active) active.classList.add('active'); });

    window.scrollTo(0, 0);
  }

  function handleInternalHashClick(e) {
    const a = e.target.closest('a[href^="#"]');
    if (!a) return;
    if (a.getAttribute('href').startsWith('javascript:')) return;
    e.preventDefault();
    const target = a.getAttribute('href');
    history.pushState(null, '', target);
    showPage(target.slice(1));
  }

  let targetAfterSplash = '#home';

  function enterApp() {
    const target = targetAfterSplash || '#home';
    const targetId = target.slice(1);
    history.pushState(null, '', target);
    showPage(targetId);
  }

  function closeSplashWithCurtain() {
    const lang = getSavedLang();
    const hint = document.querySelector('.splash-hint');
    if (hint) hint.textContent = I18N[lang].tap_to_start || 'Tap to start';

    document.body.classList.remove('splash-active');
    document.body.classList.add('splash-closing');

    const splash = document.getElementById('splash');
    if (!splash) {
      document.body.classList.remove('splash-closing');
      enterApp();
      return;
    }

    splash.classList.add('closing');

    const onEnd = (e) => {
      if (e.propertyName !== 'transform') return;
      splash.removeEventListener('transitionend', onEnd);
      splash.classList.remove('closing');
      document.body.classList.remove('splash-closing');
      enterApp();
    };

    splash.addEventListener('transitionend', onEnd);
  }

  const splashEl = document.getElementById('splash');
  if (splashEl) {
    splashEl.addEventListener('click', (e) => {
      if (e.target.closest('.lang-switch')) return;
      closeSplashWithCurtain();
    });

    splashEl.addEventListener('keydown', (e) => {
      if (e.target && e.target.closest && e.target.closest('.lang-switch')) return;
      if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault();
        closeSplashWithCurtain();
      }
    });
  }

  // ===== Tests (console.assert) =====
  const requiredPages = [
    'home','journal','journal-modern','journal-old',
    'essentials','essentials-banks','essentials-exchange','essentials-sim','essentials-taxi','essentials-pharmacy','essentials-host',
    'restaurants','restaurants-rated','restaurants-dishes','restaurants-local','restaurants-photos',
    'routes','routes-author','routes-night','routes-kayrakkum','routes-outskirts','routes-livecity','routes-phototour',
    'tourist-services','tourist-masterclass','tourist-dinner','tourist-guide','tourist-transport','tourist-allinclusive',
    'gift-store',
    'host-contacts','host-booking','host-advice','host-tips'
  ];

  ['ru','en'].forEach(l => {
    console.assert(requiredPages.every(k => k in PAGE_TITLES[l]), `PAGE_TITLES(${l}): missing keys`);
  });
  console.assert(requiredPages.every(k => k in heroBgMap), 'heroBgMap: missing keys');

  Object.entries(I18N).forEach(([l, dict]) => {
    Object.entries(dict).forEach(([k, v]) => {
      console.assert(typeof v === 'string' && v.length > 0, `I18N(${l}): empty value for ${k}`);
    });
  });

  // spot ids must be unique across modern+old
  const ids = allSpots().map(s => s.id);
  console.assert(new Set(ids).size === ids.length, 'Spot ids must be unique');

  // gift ids must be unique
  const giftIds = GIFTS.map(g => g.id);
  console.assert(new Set(giftIds).size === giftIds.length, 'Gift ids must be unique');

  // modal must find both modern and old spots
  console.assert(findSpotById('modern-1') !== null, 'findSpotById: modern spot missing');
  console.assert(findSpotById('old-1') !== null, 'findSpotById: old spot missing');

  // init (–±–µ–∑ "–º–∏–≥–∞–Ω–∏—è" ‚Äî –ø—Ä–∏–º–µ–Ω—è–µ–º —è–∑—ã–∫ –º–≥–Ω–æ–≤–µ–Ω–Ω–æ)
  applyLangInstant(getSavedLang());

  const currentHash = (location.hash && location.hash.length > 1) ? location.hash : '#home';
  targetAfterSplash = (currentHash !== '#splash') ? currentHash : '#home';

  history.replaceState(null, '', '#splash');
  showPage('splash');
